# PostgreSQL-specific requirements
# Install these only if you plan to use PostgreSQL instead of SQLite

# PostgreSQL adapter
psycopg2-binary==2.9.9

# Alternative PostgreSQL adapter (pure Python, slower but no compilation needed)
# psycopg2-cffi==2.9.0

# Note: If psycopg2-binary fails to install, you may need to:
# 1. Install PostgreSQL development headers:
#    - Ubuntu/Debian: sudo apt-get install libpq-dev python3-dev
#    - CentOS/RHEL: sudo yum install postgresql-devel python3-devel
#    - macOS: brew install postgresql
#    - Windows: Install PostgreSQL from https://www.postgresql.org/download/windows/
# 
# 2. Or use the alternative pure Python adapter:
#    pip install psycopg2-cffi
